TwitPic Javascript API Examples

media/show

// using twitpic.ready() is only required if you do not have jQuery already included in the page
twitpic.ready(function() {
	twitpic.media.show({'id':'28ymi7'}, function(media) {
		$("#media_show").html('<img src="http://twitpic.com/show/thumb/'+media.short_id+'" />');
	});
});

users/show

twitpic.users.show({'username':'meltingice'}, function(user) {
	for(var i=0; i < 5; i++) {
		$("#users_show").append('<img src="http://twitpic.com/show/thumb/'+user.images[i].short_id+'" /> ')
	}
});
