TwitPic Javascript API Examples

media/show

twitpic.media.show({'id':'28ymi7'}, function(media) {
	console.log(media);
});

users/show

twitpic.users.show({'username':'meltingice'}, function(user) {
	/* with jQuery */
	$.each(user.images, function(i,image) {
		console.log(image);
	});
	
	/* without jQuery */
	for(i in user.images) {
		console.log(user.images[i]);
	}
});
